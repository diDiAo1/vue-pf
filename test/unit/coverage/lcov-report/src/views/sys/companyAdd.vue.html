<template>
  <el-breadcrumb separator="/" class="bread">
  <el-col :span="24" class="toolbar" style="padding: 20px 0px;text-align: left">
      <el-form :inline="true">
        <el-form-item label="网站名称">
          <el-input v-model="website.websiteName" placeholder="网站名称" ></el-input>
        </el-form-item>
        <el-form-item label="网站诚信度">
          <el-input v-model="website.integrity" placeholder="网站诚信度" ></el-input>
        </el-form-item>
        <el-form-item label="公司地址">
          <el-input v-model="website.address" placeholder="公司地址" ></el-input>
        </el-form-item>
      </el-form>
      <el-form :inline="true">
        <el-form-item label="网址">
          <el-input v-model="website.url" placeholder="网址" ></el-input>
        </el-form-item>
        <el-form-item label="网站类型">
          <el-input v-model="website.webStyle" placeholder="网站类型" ></el-input>
        </el-form-item>
        <el-form-item label="备注">
          <el-input v-model="website.remark" placeholder="备注" ></el-input>
        </el-form-item>
      </el-form>
      <el-form :inline="true">
        <el-form-item>
          <el-button type="primary" @click="saveWebsite()" >保存</el-button>
        </el-form-item>
        <el-form-item>
          <el-button type="primary" class="" @click="goBack()">返回</el-button>
        </el-form-item>
      </el-form>
    </el-col>
  </el-breadcrumb>
</template>
 
<script>
export default {
    name: 'companyAdd',
    data() {
        return {
            website: {
                websiteName: '',
                integrity: '',
                address: '',
                url: 'http://www.baidu.com',
                webStyle: '1',
                remark: ''
            }
        }
    },
    methods: {
        goBack() {
            this.$router.go(-1);
        },
        saveWebsite() {
            this.$store.dispatch('addWeb',{name:this.website.websiteName,integrity:this.website.integrity,
                address:this.website.address,location:this.website.url,webStyle:this.website.webStyle,remark:this.website.remark}).then((reply)=>
                {
                    this.$router.go(-1);
                },(err) => {
                    console.log(err);
                })
        }
    }
}
</script>
